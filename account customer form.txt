using System;
					
public class Program
{
	//Application execution starts here
	
	public static void Main()
	{
		//display title
		System.Console.WriteLine("********Harsha Bank********");
		System.Console.WriteLine("::Login Page::");
		
		//declare variables to store username and passwords
		string userName = null, password = null;
		
		//read userName from Keyboard
		System.Console.WriteLine("UserName:"); 
		userName = System.Console.ReadLine();
		
		if(userName !="")
		{
		  //read password from Keyboard
		   System.Console.WriteLine("Password:"); 
		   password = System.Console.ReadLine();
		}
		
		//check userName and password
		if(userName =="system" && password =="manager")
		{
			//declare variable to store menu choice from keyborad
			int mainMenuChoice = -1;
				
			do
			{
			//displaying the main menu
		    System.Console.WriteLine("\n:: Main menu here::");
			System.Console.WriteLine("1.Customer");
			System.Console.WriteLine("2.accounts");

			System.Console.WriteLine("4.fund Transfer stmt");
			System.Console.WriteLine("5.Account stmt");
			System.Console.WriteLine("0.exit");
			
			System.Console.Write("Enter choice");
			mainMenuChoice = int.Parse(System.Console.ReadLine());
			
			//switch-case to check menu choice
			switch(mainMenuChoice)
			{
				case 1:CustomersMenu();break;
				case 2:AccountMenu();break;
				case 3://To DO:Display fund transfer
				    break;
				case 4://To DO:Display fund transfer stmt
				    break;
				case 5://To DO:Display account stmt
				    break;
				case 0://To DO:Display exit

			}
		  }while(mainMenuChoice !=0);	
		}
		else
		{
			//if wrong username or password
			System.Console.WriteLine("invalid userName or passord");
		}	
		
		//about to exit(thanks msg to user)
		System.Console.WriteLine("Thank you! visit again");
		System.Console.ReadKey();

			System.Console.WriteLine("\n::::customers menu:::");

			System.Console.WriteLine("3.update customer");
			System.Console.WriteLine("4.view customer");
